const Max_out_of_three = (A,B,C) => 
        {
          if((A==B) && (A==C))
          {
            return -1;
          }
          
          if((A>B) && (A>C))
          {
            return A;
          }
          if((B>C) && (B>A))
          {
            return B;
          }
          if((C>A) && (C>B))
          {
            return C;
          }
